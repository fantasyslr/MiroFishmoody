**run.py 实现要点：**

1. **InMemoryTaskManager** — EvaluationOrchestrator 需要 TaskManager，创建一个最小 in-memory 实现（不依赖 SQLite），避免 runner 需要数据库：
```python
class _InMemoryTaskManager:
    def __init__(self): self._tasks = {}
    def create_task(self, task_id): self._tasks[task_id] = {"status": "pending", "progress": 0}
    def update_task(self, task_id, status=None, progress=None, message=None):
        if task_id in self._tasks:
            if status: self._tasks[task_id]["status"] = status
            if progress is not None: self._tasks[task_id]["progress"] = progress
    def get_task(self, task_id): return self._tasks.get(task_id)
```

2. **mock LLMClient 注入** — 用 `unittest.mock.patch` 在 EvaluationOrchestrator.run() 调用期间将 `backend.app.utils.llm_client.LLMClient` patch 为 MockLLMClient：

from collections import defaultdict
hits = defaultdict(int)
totals = defaultdict(int)

for fixture_path in sorted(glob.glob(fixture_dir)):
    data, campaign_set = _load_fixture(fixture_path)
    brief_type_str = data["brief_type"]
    brief_type_enum = BriefType(brief_type_str)
    category = data["campaigns"][0]["product_line"]  # 用第一个 campaign 的 product_line 作 category

    task_id = f"bench_{data['id']}"

    with patch('backend.app.services.evaluation_orchestrator.LLMClient', MockLLMClient):
        result_dict = orchestrator.run(task_id, campaign_set, category=category, brief_type=brief_type_enum)

    # result_dict["rankings"] 已按 rank 升序排列，[0] 是冠军
    predicted_winner = result_dict["rankings"][0]["campaign_id"]

    totals[brief_type_str] += 1
    if predicted_winner == expected_winner:
        hits[brief_type_str] += 1
