Controversy detection logic (pure frontend computation, no API change needed):
```typescript
// For each campaign_id, check:
// 1. stdev of panel_scores for this campaign >= 2.0 (matches ConsensusAgent threshold)
// 2. OR any pairwise vote for this campaign has dissent: true

function isControversial(
  campaignId: string,
  panelScores: EvalPanelScore[],
  pairwiseResults: EvalPairwiseResult[]
): boolean {
  // Check suspect flag from ConsensusAgent
  const hasSuspect = panelScores
    .filter(ps => ps.campaign_id === campaignId)
    .some(ps => ps.dimension_scores?.suspect === true)

  // Check dissent votes from devil's advocate
  const hasDissent = pairwiseResults.some(pr =>
    (pr.campaign_a_id === campaignId || pr.campaign_b_id === campaignId) &&
    pr.votes.some(v => v.dissent === true)
  )

  return hasSuspect || hasDissent
}
```
