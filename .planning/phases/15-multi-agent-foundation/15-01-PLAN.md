    def __init__(self, llm_client=None, max_workers=3):
        self.llm = llm_client or LLMClient()
        self.max_workers = max_workers
        self._semaphore = threading.Semaphore(max_workers)  # ← 要删除

    def _safe_analyze_single(self, image_path):
        self._semaphore.acquire()  # ← 要删除
        try:
            return self.analyze_single_image(image_path)
        finally:
            self._semaphore.release()  # ← 要删除
